
Namespace example:
<people xmlns:xsi="a" xsi:schemaLocation="http://..." xmlns:l="b" xmlns="c"> 
	<person>
		<name>Marcus</name>
		<language>
			<l:primary>XML</l:primary>
			<xsi:secondary>HTML</xsi:secondary>
		</language>
	</person>
</people>
toMap (keep root): Array
(
    [people] => Array
        (
            [person] => Array
                (
                    [name] => Marcus
                    [language] => Array
                        (
                            [xsi:secondary] => HTML
                            [l:primary] => XML
                        )

                )

        )

)
