Simple example:
<doc
	language="en">
	<t>t1</t>
	<br/>
	<t>t2</t>
	<br/>
	<s>
		<sa>sa</sa>
		<sb x="xx" t="true" />
		sval
	</s>
</doc>
toMap: Array
(
    [@] => Array
        (
            [language] => en
        )

    [t.0] => t1
    [t.1] => t2
    [br.0] => 
    [br.1] => 
    [s] => Array
        (
            [sa] => sa
            [sb] => Array
                (
                    [@] => Array
                        (
                            [x] => xx
                            [t] => true
                        )

                )

            [=] => sval
        )

)
